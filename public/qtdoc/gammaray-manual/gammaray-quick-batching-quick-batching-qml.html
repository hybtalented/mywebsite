import QtQuick 2.0
import QtQuick.Window 2.0

Window {
    id: myWindow
    height: 600
    width: 100

    Row {
        id: myRow
        anchors.fill: parent
        anchors.margins: spacing
        spacing: 5

        Slider {
            id: leftSlider
            anchors.top: parent.top
            anchors.bottom: parent.bottom
            width: myWindow.width / 2 - 1.5*myRow.spacing
        }

        Slider {
            id: rightSlider
            anchors.top: parent.top
            anchors.bottom: parent.bottom
            width: myWindow.width / 2 - 1.5*myRow.spacing
            mirrorSlider: true
        }
    }
}
